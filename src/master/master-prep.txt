# Execute update/upgrade

# Execute worker install script 2 (disable swap, install kubernetes, initializing kubernetes)
# Note: make sure you modify the script to use your master node IP address

# Execute kubeadm join command to join the node to the cluster

# On the master, execute the kubectl label command to define the new node as a worker
kubectl label node kub-worker-01 node-role.kubernetes.io/worker=worker # only change: 'kub-worker-01' should be your host name of the node
